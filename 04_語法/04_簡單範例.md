## 一般寫法

1. 一個輸出時間的簡單範例。

    ```cpp
    // 定義時間相關變量
    unsigned long previousMillis = 0;
    const long interval = 1000; // 1 秒
    int seconds = 0;
    int minutes = 0;
    int hours = 0;
    int days = 0;

    void setup() {
        Serial.begin(115200); // 初始化序列通信
        Serial.println("Arduino Digital Clock");
    }

    void loop() {
        unsigned long currentMillis = millis();

        // 每秒更新一次時間
        if (currentMillis - previousMillis >= interval) {
            previousMillis = currentMillis;
            seconds++;

            // 處理秒數增加
            if (seconds >= 60) {
            seconds = 0;
            minutes++;
            }

            // 處理分鐘數增加
            if (minutes >= 60) {
            minutes = 0;
            hours++;
            }

            // 處理小時數增加
            if (hours >= 24) {
            hours = 0;
            days++;
            }

            // 顯示時間
            displayTime();
        }
    }

    void displayTime() {
        Serial.print("Time: ");
        Serial.print(days);
        Serial.print(" days ");
        Serial.print(hours);
        Serial.print(":");
        Serial.print(minutes);
        Serial.print(":");
        Serial.println(seconds);
    }
    ```

## 改用物件導向

1. 先統一寫在主腳本中。

    ```cpp
    // 定義 Clock 類
    class Clock {
    public:
        Clock() {
            previousMillis = 0;
            seconds = 0;
            minutes = 0;
            hours = 0;
            days = 0;
        }

        void update(unsigned long currentMillis, unsigned long interval) {
            if (currentMillis - previousMillis >= interval) {
                previousMillis = currentMillis;
                seconds++;

                if (seconds >= 60) {
                    seconds = 0;
                    minutes++;
                }

                if (minutes >= 60) {
                    minutes = 0;
                    hours++;
                }

                if (hours >= 24) {
                    hours = 0;
                    days++;
                }
            }
        }

        void display() {
            Serial.print("Time: ");
            Serial.print(days);
            Serial.print(" days ");
            Serial.print(hours);
            Serial.print(":");
            Serial.print(minutes);
            Serial.print(":");
            Serial.println(seconds);
        }

    private:
        unsigned long previousMillis;
        int seconds;
        int minutes;
        int hours;
        int days;
    };

    // 創建 Clock 對象
    Clock clock;

    // 設置更新間隔為 1 秒
    const unsigned long interval = 1000;

    void setup() {
        // 初始化串行通信
        Serial.begin(115200);
        // 初始延遲以確保串行通信穩定
        delay(1000);
        while(!Serial){
            delay(50);
        }
        Serial.println("Arduino Digital Clock with Class");
    }

    void loop() {
        unsigned long currentMillis = millis();
        clock.update(currentMillis, interval);
        clock.display();
        // 每秒更新一次顯示
        delay(1000);
    }
    ```
